  157.     public function save() {

  159.             throw new exception(get_class($this) . ': Can\'t save table row inside template view!');
  160.         }

  182.         
  183.         $old_id = $id;
  184.         
  185.         if (is_null($id)) {

  193.             if (!$this->db->update($this->table_name)) {
  194.                 return FALSE;

  234.     /**
  235.      * Multipurpose function, based on given arguments.
  236.      * 
  237.      * If you omit both arguments, this function returns array of original data from table row.
  238.      * If you specify first argument as string and leaves second intact, you will receive value of given column from table row.
  239.      * If you specify first argument as array of string => string | bool | numeric | Nonescape_data and leaves second intact,
  240.      * you will set new values of table column from given array and function returns reference to this object.
  241.      * If you specify both arguments, first must be string and second can be string | bool | numeric | Nonescape_data, you will
  242.      * set new value of table column for given column and value, and you will receive reference to this object.
  243.      * 
  244.      * Any other combination of arguments can trigger error or do nothing, just return reference to this object.
  245.      * 
  246.      * @param NULL|string|array<mixed>$column nothing, column name or array of new column values.
  247.      * @param NULL|string|bool|numeric|Nonescape_data$new_value nothing or new value of column.
  248.      * @return mixed can return array of original data, value of column, NULL or reference to this object.
  249.      */
  250.     public function data($column = NULL, $new_value = NULL) {
  251.         if (is_null($column) && is_null($new_value)) {
  252.             return $this->original_data;
  253.         } else if (!is_null($column) && is_null($new_value)) {
  254.             if (is_string($column)) {
  255.                 return isset($this->original_data[$column]) ? $this->original_data[$column] : (!in_array($column, $this->_getKnownFields()) && method_exists($this, 'get' . $column) ? call_user_func(array($this, 'get' . $column)) : NULL);    
  256.             } else if (is_array($column)) {
  257.                 if (count($column)) {
  258.                     foreach ($column as $field => $value) {
  259.                         $this->data($field, $value);
  260.                     }
  261.                 }
  262.             }
  263.         } else if (!is_null($column) && !is_null($new_value)) {
  264.             if (is_string($column)) {
  265.                 if (in_array($column, $this->_getKnownFields())) { 
  266.                     if (is_string($new_value) || is_bool($new_value) || is_numeric($new_value) || $new_value instanceof Nonescape_data) {
  267.                         if ($this->isInsideTemplate()) {
  268.                             throw new exception(get_class($this) . ': Can\'t set table row column value inside template view!');
  269.                         }
  270.                         if ($column == $this->primary_field && isset($this->original_data[$this->primary_field])) { 
  271.                             throw new exception(get_class($this) . ': Can\'t change existing id from ' . $this->original_data[$this->primary_field] . ' to ' . $new_value . '!');
  272.                         } else if ($column == 'tstamp') {
  273.                             throw new exception(get_class($this) . ': Can\'t change time of last modification!');
  274.                         } else if ($column == 'crdate') {
  275.                             throw new exception(get_class($this) . ': Can\'t change time of record creation!');
  276.                         }
  277.                         
  278.                         if ($column == $this->primary_field) {
  279.                             $old_data = $this->data;
  280.                             if (!$this->load($new_value)) {
  281.                                 throw new exception(get_class($this) . ': There is no primary key ' . $new_value . ' in table ' . $this->table_name . '!');
  282.                             }
  283.                             $this->data = $old_data;
  284.                             return $this;   
  285.                         }
  286.                         
  287.                         $this->data[$column] = $new_value;
  288.                     } else {
  289.                         throw new exception(get_class($this) . ': Can\'t handle supplied value! It is of wrong type. Allowed types are: string, boolean, numeric and Nonescape_data');
  290.                     }
  291.                 } /*else {
  292.                     throw new exception(get_class($this) . ': Unknown column "' . $column . '" in table ' . $this->table_name . '!');
  293.                 }*/
  294.             } else {
  295.                 throw new exception(get_class($this) . ': Column name must be string!');
  296.             }
  297.         }
  298.         return $this;
