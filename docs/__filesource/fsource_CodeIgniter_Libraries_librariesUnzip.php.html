  263.     private function _extract_file($compressed_file_name, $target_file_name = FALSE)

  266.         {

  268.             $this->_list_files(FALSE, $compressed_file_name);

  279.         if (substr($compressed_file_name, -1) == '/')

  284.  
  285.         if ( ! $fdetails['uncompressed_size'])

  291.  
  292.         fseek($this->fh, $fdetails['contents_start_offset']);
  293.         $ret = $this->_uncompress(
  294.             fread($this->fh, $fdetails['compressed_size']),
  295.             $fdetails['compression_method'],
  296.             $fdetails['uncompressed_size'],
  297.             $target_file_name
  298.         );

  300.         if ($this->apply_chmod AND $target_file_name)

  305.         return $ret;

  395.     private function _load_file_list_by_eof(&$fh, $stop_on_file = FALSE)
  396.     {
  397.         // Check if there's a valid Central Dir signature.
  398.         // Let's consider a file comment smaller than 1024 characters...
  399.         // Actually, it length can be 65536.. But we're not going to support it.
  400.  
  401.         for ($x = 0; $x < 1024; $x++)
  402.         {
  403.             fseek($fh, -22 - $x, SEEK_END);
  404.  
  405.             $signature = fread($fh, 4);
  406.  
  407.             if ($signature == $this->central_signature_end)
  408.             {
  409.                 // If found EOF Central Dir
  410.                 $eodir['disk_number_this'] = unpack("v", fread($fh, 2)); // number of this disk
  411.                 $eodir['disk_number'] = unpack("v", fread($fh, 2)); // number of the disk with the start of the central directory
  412.                 $eodir['total_entries_this'] = unpack("v", fread($fh, 2)); // total number of entries in the central dir on this disk
  413.                 $eodir['total_entries'] = unpack("v", fread($fh, 2)); // total number of entries in
  414.                 $eodir['size_of_cd'] = unpack("V", fread($fh, 4)); // size of the central directory
  415.                 $eodir['offset_start_cd'] = unpack("V", fread($fh, 4)); // offset of start of central directory with respect to the starting disk number
  416.                 $zip_comment_lenght = unpack("v", fread($fh, 2)); // zipfile comment length
  417.                 $eodir['zipfile_comment'] = $zip_comment_lenght[1] ? fread($fh, $zip_comment_lenght[1]) : ''; // zipfile comment
  418.  
  419.                 $this->end_of_central = array(
  420.                     'disk_number_this' => $eodir['disk_number_this'][1],
  421.                     'disk_number' => $eodir['disk_number'][1],
  422.                     'total_entries_this' => $eodir['total_entries_this'][1],
  423.                     'total_entries' => $eodir['total_entries'][1],
  424.                     'size_of_cd' => $eodir['size_of_cd'][1],
  425.                     'offset_start_cd' => $eodir['offset_start_cd'][1],
  426.                     'zipfile_comment' => $eodir['zipfile_comment'],
  427.                 );
  428.  
  429.                 // Then, load file list
  430.                 fseek($fh, $this->end_of_central['offset_start_cd']);
  431.                 $signature = fread($fh, 4);
  432.  
  433.                 while ($signature == $this->dir_signature)
  434.                 {
  435.                     $dir['version_madeby'] = unpack("v", fread($fh, 2)); // version made by
  436.                     $dir['version_needed'] = unpack("v", fread($fh, 2)); // version needed to extract
  437.                     $dir['general_bit_flag'] = unpack("v", fread($fh, 2)); // general purpose bit flag
  438.                     $dir['compression_method'] = unpack("v", fread($fh, 2)); // compression method
  439.                     $dir['lastmod_time'] = unpack("v", fread($fh, 2)); // last mod file time
  440.                     $dir['lastmod_date'] = unpack("v", fread($fh, 2)); // last mod file date
  441.                     $dir['crc-32'] = fread($fh, 4);              // crc-32
  442.                     $dir['compressed_size'] = unpack("V", fread($fh, 4)); // compressed size
  443.                     $dir['uncompressed_size'] = unpack("V", fread($fh, 4)); // uncompressed size
  444.                     $zip_file_length = unpack("v", fread($fh, 2)); // filename length
  445.                     $extra_field_length = unpack("v", fread($fh, 2)); // extra field length
  446.                     $fileCommentLength = unpack("v", fread($fh, 2)); // file comment length
  447.                     $dir['disk_number_start'] = unpack("v", fread($fh, 2)); // disk number start
